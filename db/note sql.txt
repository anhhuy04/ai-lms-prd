-- ─── 1.1 PROFILES ───
-- Hồ sơ người dùng, tự động tạo khi auth.users insert (trigger)
CREATE TABLE IF NOT EXISTS public.profiles (
  id          uuid PRIMARY KEY REFERENCES auth.users(id),  -- ID = auth.users.id
  full_name   text,                                         -- Họ tên đầy đủ
  role        text DEFAULT 'student'                        -- Vai trò: 'teacher' | 'student' | 'admin'
              CHECK (role IN ('teacher', 'student', 'admin')),
  avatar_url  text,                                         -- URL ảnh đại diện (Supabase Storage)
  bio         text,                                         -- Giới thiệu ngắn
  phone       text,                                         -- Số điện thoại
  gender      text                                          -- Giới tính: 'male' | 'female' | 'other'
              CHECK (gender IN ('male', 'female', 'other')),
  metadata    jsonb,                                        -- Metadata mở rộng
  updated_at  timestamptz DEFAULT now()
);
-- JSONB: profiles.metadata
-- {
--   "preferences": { "theme": "dark", "language": "vi" },
--   "school_info": { "student_code": "HS001", "class_name": "10A1" }
-- }


-- ─── 1.2 SCHOOLS ───
-- Trường học - container cấp cao nhất
CREATE TABLE IF NOT EXISTS public.schools (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name        text NOT NULL,                                -- Tên trường
  domain      text UNIQUE,                                  -- Domain email (vd: "school.edu.vn")
  metadata    jsonb,                                        -- Thông tin mở rộng về trường
  created_at  timestamptz DEFAULT now()
);
-- JSONB: schools.metadata
-- { "address": "123 ABC", "phone": "028-1234", "logo_url": "https://..." }


-- ─── 1.3 CLASSES ───
-- Lớp học - đơn vị quản lý chính
CREATE TABLE IF NOT EXISTS public.classes (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  school_id       uuid REFERENCES public.schools(id),       -- Thuộc trường nào (nullable = lớp độc lập)
  teacher_id      uuid NOT NULL REFERENCES auth.users(id),  -- Giáo viên chủ nhiệm
  name            text NOT NULL,                             -- Tên lớp (vd: "Toán 10A1")
  subject         text,                                      -- Môn học
  academic_year   text,                                      -- Năm học (vd: "2025-2026")
  description     text,                                      -- Mô tả lớp
  class_settings  jsonb,                                     -- Cấu hình lớp (xem mẫu JSONB)
  created_at      timestamptz DEFAULT now()
);
-- JSONB: classes.class_settings (default value)
-- {
--   "defaults": { "lock_class": false },
--   "enrollment": {
--     "qr_code": {
--       "is_active": false,     -- QR code hoạt động?
--       "join_code": null,      -- Mã tham gia
--       "expires_at": null,     -- Hạn QR
--       "require_approval": true -- Cần GV duyệt
--     },
--     "manual_join_limit": null  -- Giới hạn HS (null = ko giới hạn)
--   },
--   "group_management": {
--     "lock_groups": false,
--     "allow_student_switch": false,
--     "is_visible_to_students": true
--   },
--   "student_permissions": {
--     "auto_lock_on_submission": false,
--     "can_edit_profile_in_class": true
--   }
-- }


-- ─── 1.4 CLASS_MEMBERS ───
-- Danh sách học sinh trong lớp
CREATE TABLE IF NOT EXISTS public.class_members (
  class_id    uuid NOT NULL REFERENCES public.classes(id),
  student_id  uuid NOT NULL REFERENCES auth.users(id),
  role        text DEFAULT 'student',                        -- Vai trò trong lớp
  joined_at   timestamptz DEFAULT now(),                     -- Thời điểm tham gia
  status      text DEFAULT 'pending'                         -- 'pending' | 'approved' | 'rejected'
              CHECK (status IN ('pending', 'approved', 'rejected')),
  PRIMARY KEY (class_id, student_id)
);


-- ─── 1.5 CLASS_TEACHERS ───
-- Giáo viên phụ trách lớp (hỗ trợ nhiều GV/lớp)
CREATE TABLE IF NOT EXISTS public.class_teachers (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  class_id    uuid REFERENCES public.classes(id),
  teacher_id  uuid REFERENCES auth.users(id),
  role        text DEFAULT 'teacher'                         -- 'teacher' | 'assistant'
);


-- ─── 1.6 GROUPS ───
-- Nhóm học sinh trong lớp
CREATE TABLE IF NOT EXISTS public.groups (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  class_id    uuid REFERENCES public.classes(id),            -- Thuộc lớp nào
  teacher_id  uuid REFERENCES auth.users(id),                -- GV tạo nhóm (legacy)
  name        text NOT NULL,                                  -- Tên nhóm
  description text,
  created_at  timestamptz DEFAULT now()
);


-- ─── 1.7 GROUP_MEMBERS ───
-- Thành viên trong nhóm
CREATE TABLE IF NOT EXISTS public.group_members (
  group_id    uuid NOT NULL REFERENCES public.groups(id),
  student_id  uuid NOT NULL REFERENCES public.profiles(id),  -- FK → profiles
  role        text DEFAULT 'member',                          -- 'member' | 'leader'
  enrolled_by uuid REFERENCES auth.users(id),                 -- Ai thêm vào nhóm
  joined_at   timestamptz DEFAULT now(),
  PRIMARY KEY (group_id, student_id)
);

-- ─── 2.1 LEARNING_OBJECTIVES ───
-- Mục tiêu học tập (chuẩn kiến thức) - cấu trúc cây
CREATE TABLE IF NOT EXISTS public.learning_objectives (
  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  subject_code  text NOT NULL,                               -- Mã môn (vd: "MATH")
  code          text NOT NULL,                               -- Mã chuẩn KT (vd: "M3.1.1")
  description   text NOT NULL,                               -- Mô tả mục tiêu
  difficulty    integer CHECK (difficulty >= 1 AND difficulty <= 5),
  parent_id     uuid REFERENCES public.learning_objectives(id),
  metadata      jsonb,
  created_at    timestamptz DEFAULT now()
);
-- JSONB: learning_objectives.metadata
-- { ... }


-- ─── 2.2 QUESTIONS ───
-- Ngân hàng câu hỏi (Question Bank) - câu hỏi gốc, tái sử dụng
CREATE TABLE IF NOT EXISTS public.questions (
  id              uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  author_id       uuid NOT NULL REFERENCES auth.users(id),   -- GV tạo câu hỏi
  type            text NOT NULL                               -- Loại câu hỏi
                  CHECK (type IN (
                    'multiple_choice', 'short_answer', 'essay', 'true_false',
                    'matching', 'problem_solving', 'file_upload', 'fill_in_blank'
                  )),
  content         jsonb NOT NULL,                             -- Nội dung câu hỏi
  answer          jsonb,                                      -- Đáp án đúng
  default_points  numeric DEFAULT 1 CHECK (default_points > 0),
  difficulty      integer CHECK (difficulty >= 1 AND difficulty <= 5),
  tags            text[],                                     -- Tags phân loại
  is_public       boolean DEFAULT false,                      -- Câu hỏi công khai
  created_at      timestamptz DEFAULT now(),
  updated_at      timestamptz DEFAULT now()
);

-- ─── 2.3 QUESTION_CHOICES ───
-- Đáp án câu hỏi trong ngân hàng
-- LƯU Ý: id là INTEGER (không UUID) → tiết kiệm, index nhanh
CREATE TABLE IF NOT EXISTS public.question_choices (
  id          integer NOT NULL CHECK (id >= 0),              -- 0, 1, 2, 3...
  question_id uuid NOT NULL REFERENCES public.questions(id),
  content     jsonb NOT NULL,                                 -- BẮT BUỘC là Object
  is_correct  boolean DEFAULT false,                          -- Đáp án đúng?
  PRIMARY KEY (id, question_id)
);
-- ╔═══════════════════════════════════════════════════════════════════╗
-- ║  JSONB: question_choices.content                               ║
-- ║  ⚠️ BẮT BUỘC là Object, có mồi cho AI (tránh crash map())   ║
-- ╚═══════════════════════════════════════════════════════════════════╝
-- { "text": "Đáp án A", "image_url": null, "distractor_rationale": "Lý do HS hay chọn sai..." }


-- ─── 2.4 QUESTION_OBJECTIVES ───
-- Liên kết N-N: câu hỏi ↔ mục tiêu
CREATE TABLE IF NOT EXISTS public.question_objectives (
  question_id   uuid NOT NULL REFERENCES public.questions(id),
  objective_id  uuid NOT NULL REFERENCES public.learning_objectives(id),
  PRIMARY KEY (question_id, objective_id)
);


-- ─── 2.5 ASSIGNMENTS ───
-- Template bài tập (KHÔNG có deadline/time_limit → nằm ở distributions)
CREATE TABLE IF NOT EXISTS public.assignments (
  id                        uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  class_id                  uuid REFERENCES public.classes(id),
  teacher_id                uuid NOT NULL REFERENCES auth.users(id),
  title                     text NOT NULL,                              -- Tiêu đề
  description               text,                                       -- Mô tả / ghi chú
  is_published              boolean DEFAULT false,
  published_at              timestamptz,
  total_points              numeric CHECK (total_points IS NULL OR total_points >= 0),
  default_shuffle_questions boolean DEFAULT false,                      -- Đảo câu hỏi
  default_shuffle_choices   boolean DEFAULT false,                      -- Đảo đáp án
  created_at                timestamptz DEFAULT now(),
  updated_at                timestamptz DEFAULT now()
);


-- ─── 2.6 ASSIGNMENT_QUESTIONS ───
-- Câu hỏi trong bài tập
-- question_id = NULL → inline (custom_content chứa toàn bộ)
-- question_id != NULL → link ngân hàng, custom_content = override
CREATE TABLE IF NOT EXISTS public.assignment_questions (
  id               uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  assignment_id    uuid NOT NULL REFERENCES public.assignments(id),
  question_id      uuid REFERENCES public.questions(id),     -- NULL = inline
  custom_content   jsonb,                                     -- Delta Override
  points           numeric NOT NULL DEFAULT 1 CHECK (points > 0),
  rubric           jsonb,                                     -- Rubric chấm điểm
  order_idx        integer NOT NULL                           -- Thứ tự (1, 2, 3...)
);
-- ╔═══════════════════════════════════════════════════════════════════╗
-- ║  JSONB: assignment_questions.custom_content (DELTA OVERRIDE)     ║
-- ║  ⚠️ TỬ HUYỆT 2: Chỉ lưu khác biệt so với questions gốc        ║
-- ╚═══════════════════════════════════════════════════════════════════╝
--
-- Chỉ override_text:
-- { "override_text": "Nội dung câu hỏi đã sửa cho lớp 10A" }
--
-- Override cả đáp án:
-- {
--   "override_text": "Nội dung đã sửa",
--   "choices": [
--     {"id": 0, "text": "Đáp án A đã sửa", "isCorrect": true},
--     {"id": 1, "text": "Đáp án B", "isCorrect": false}
--   ]
-- }
--
-- Essay với keywords:
-- { "override_text": "Viết về...", "expected_answer": "Mẫu", "ai_grading_keywords": [...] }

-- ╔═══════════════════════════════════════════════════════════════════╗
-- ║  JSONB: assignment_questions.rubric (MA TRẬN CHẤM ĐIỂM)        ║

-- ─── 2.7 ASSIGNMENT_VARIANTS ───
-- Biến thể đề (shuffle câu hỏi/đáp án cho từng HS/nhóm)
CREATE TABLE IF NOT EXISTS public.assignment_variants (

  variant_type      text NOT NULL DEFAULT 'student'
                    CHECK (variant_type IN ('student', 'group', 'global')),
  student_id        uuid REFERENCES auth.users(id),
  group_id          uuid REFERENCES public.groups(id),
  due_at_override   timestamptz,                             -- Override deadline riêng

-- ─── 2.8 ASSIGNMENT_DISTRIBUTIONS ───
-- Đợt phát hành = NGUỒN SỰ THẬT cho deadline, time_limit, status
CREATE TABLE IF NOT EXISTS public.assignment_distributions (
  id                  uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  assignment_id       uuid NOT NULL REFERENCES public.assignments(id),
  distribution_type   text NOT NULL
                      CHECK (distribution_type IN ('class', 'group', 'individual')),
  class_id            uuid REFERENCES public.classes(id),    -- Lớp nhận
  group_id            uuid REFERENCES public.groups(id),     -- Nhóm nhận
  student_ids         uuid[],                                 -- DS học sinh (individual)
  available_from      timestamptz,                            -- Bắt đầu từ
  due_at              timestamptz,                            -- Hạn nộp
  time_limit_minutes  integer CHECK (time_limit_minutes IS NULL OR time_limit_minutes > 0),
  allow_late          boolean DEFAULT true,
  late_policy         jsonb,                                  -- Chính sách nộp trễ
  status              text NOT NULL DEFAULT 'active'
                      CHECK (status IN ('draft', 'scheduled', 'active', 'closed', 'archived')),
  settings            jsonb DEFAULT '{"shuffle_choices": false, "shuffle_questions": false, "show_score_immediately": true, "student_review_mode": "full_review", "ai_feedback_enabled": true}',
  created_at          timestamptz DEFAULT now()
);

-- ─── 3.1 WORK_SESSIONS ───
-- Phiên làm bài (tracking từ mở đề → nộp)
CREATE TABLE IF NOT EXISTS public.work_sessions (
  id                        uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  assignment_distribution_id uuid REFERENCES public.assignment_distributions(id),
  assignment_id             uuid REFERENCES public.assignments(id),
  student_id                uuid NOT NULL REFERENCES public.profiles(id),
  started_at                timestamptz,                     -- Bắt đầu làm
  submitted_at              timestamptz,                     -- Nộp bài
  attempt                   integer DEFAULT 1,               -- Lần thứ mấy
  status                    text DEFAULT 'in_progress'
                            CHECK (status IN ('in_progress', 'submitted', 'graded')),
  time_spent_seconds        bigint DEFAULT 0,                -- Thời gian làm (giây)
  created_at                timestamptz DEFAULT now(),
  updated_at                timestamptz DEFAULT now()
);


-- ─── 3.2 AUTOSAVE_ANSWERS ───
-- Lưu tự động (draft) khi HS đang làm bài
CREATE TABLE IF NOT EXISTS public.autosave_answers (
  id                      uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id              uuid REFERENCES public.work_sessions(id),
  assignment_question_id  uuid REFERENCES public.assignment_questions(id),
  answer_content          jsonb,                             -- Nội dung draft
  updated_at              timestamptz DEFAULT now()
);
-- JSONB: autosave_answers.answer_content
-- Multiple choice: { "selected_choices": [0] }
-- Essay:           { "text": "Bài làm..." }


-- ─── 3.3 SUBMISSIONS ───
-- Bài nộp chính thức
CREATE TABLE IF NOT EXISTS public.submissions (
  id                        uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  assignment_id             uuid REFERENCES public.assignments(id),
  assignment_distribution_id uuid REFERENCES public.assignment_distributions(id),
  student_id                uuid NOT NULL REFERENCES public.profiles(id),
  session_id                uuid REFERENCES public.work_sessions(id),
  variant_id                uuid REFERENCES public.assignment_variants(id),
  started_at                timestamptz,
  submitted_at              timestamptz,
  is_late                   boolean DEFAULT false,           -- Nộp trễ? (boolean tĩnh)
  total_score               numeric,                         -- Tổng điểm
  ai_graded                 boolean DEFAULT false,           -- AI đã chấm?
  is_voided                 boolean DEFAULT false,           -- Bài bị hủy
  created_at                timestamptz DEFAULT now(),
  updated_at                timestamptz DEFAULT now()
);


-- ─── 3.4 SUBMISSION_ANSWERS ───
-- Câu trả lời chính thức từng câu
CREATE TABLE IF NOT EXISTS public.submission_answers (
  id                      uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id              uuid REFERENCES public.work_sessions(id),
  assignment_question_id  uuid NOT NULL REFERENCES public.assignment_questions(id),
  answer                  jsonb NOT NULL,                    -- Câu trả lời
  files                   jsonb,                             -- Files đính kèm
  flagged                 boolean DEFAULT false,             -- GV đánh dấu review
  ai_score                numeric,                           -- Điểm AI
  ai_confidence           numeric,                           -- Độ tin cậy AI (0-1)
  final_score             numeric,                           -- Điểm cuối cùng
  ai_feedback             jsonb,                             -- Phản hồi AI
  teacher_feedback        jsonb,                             -- Phản hồi GV
  graded_by               uuid REFERENCES auth.users(id),   -- Ai chấm (null=AI)
  graded_at               timestamptz,
  created_at              timestamptz DEFAULT now(),
  updated_at              timestamptz DEFAULT now()
);
-- JSONB: submission_answers.answer
-- Multiple choice: { "selected_choices": [0] }     ← Mảng INT
-- True/false:      { "selected_choices": [0] }     ← 0=true, 1=false
-- Essay:           { "text": "Bài làm..." }
--
-- JSONB: submission_answers.ai_feedback
-- { "comment": "...", "strengths": [], "weaknesses": [], "suggestions": [] }
--
-- JSONB: submission_answers.teacher_feedback
-- { "comment": "...", "rating": 3 }
--
-- JSONB: submission_answers.files
-- [{ "file_id": "uuid", "name": "bai_lam.pdf", "url": "https://..." }]


-- ─── 3.5 FILES ───
-- File upload (Supabase Storage)
CREATE TABLE IF NOT EXISTS public.files (
  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  storage_path  text NOT NULL,                               -- Path trong Storage bucket
  url           text,                                         -- Public URL
  filename      text,                                         -- Tên file gốc
  mime_type     text,                                         -- MIME type
  size_bytes    bigint,                                       -- Kích thước (bytes)
  uploaded_by   uuid REFERENCES auth.users(id),
  metadata      jsonb,
  is_deleted    boolean DEFAULT false,                        -- Soft delete
  created_at    timestamptz DEFAULT now()
);


-- ─── 3.6 FILE_LINKS ───
-- Liên kết polymorphic: file ↔ entity
CREATE TABLE IF NOT EXISTS public.file_links (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  file_id     uuid REFERENCES public.files(id),
  target_type text NOT NULL,                                  -- 'assignment' | 'submission' | ...
  target_id   uuid NOT NULL,
  created_at  timestamptz DEFAULT now()
);


-- ─── 3.7 AI_QUEUE ───
-- Hàng đợi chấm bài AI (async)
CREATE TABLE IF NOT EXISTS public.ai_queue (
  id                    uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  submission_answer_id  uuid REFERENCES public.submission_answers(id),
  request_type          text DEFAULT 'score'
                        CHECK (request_type IN ('score', 'feedback', 'analysis')),
  status                text DEFAULT 'pending',              -- pending | processing | completed | failed
  attempts              integer DEFAULT 0,
  payload               jsonb,                               -- Data gửi AI
  result                jsonb,                               -- Kết quả AI trả về
  created_at            timestamptz DEFAULT now(),
  updated_at            timestamptz DEFAULT now()
);
-- JSONB: ai_queue.payload
-- { "question_text": "...", "answer_text": "...", "rubric": {...}, "max_points": 5 }


-- ─── 3.8 AI_EVALUATIONS ───
-- Kết quả đánh giá AI (lịch sử)
CREATE TABLE IF NOT EXISTS public.ai_evaluations (
  id                    uuid PRIMARY KEY DEFAULT gen_random_uuid(),

  model_name            text,                                -- "gpt-4" | "gemini-pro"
  model_version         text,
  ai_score              numeric,
  ai_confidence         numeric,                             -- 0.0 - 1.0
  feedback              text,
  rationale             jsonb,                               -- Giải thích chi tiết
  created_at            timestamptz DEFAULT now()
);
-- JSONB: ai_evaluations.rationale
-- {

--   "overall_comment": "..."

-- ─── 3.9 GRADE_OVERRIDES ───
-- Audit trail: GV chỉnh điểm
CREATE TABLE IF NOT EXISTS public.grade_overrides (
  id                    uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  submission_answer_id  uuid NOT NULL REFERENCES public.submission_answers(id),
  overridden_by         uuid NOT NULL REFERENCES auth.users(id),
  old_score             numeric,
  new_score             numeric NOT NULL,
  reason                text,
  created_at            timestamptz NOT NULL DEFAULT now()
);


-- ─── 3.10 STUDENT_SKILL_MASTERY ───
-- Trình độ kỹ năng HS (AI tính toán)
CREATE TABLE IF NOT EXISTS public.student_skill_mastery (
  student_id    uuid NOT NULL REFERENCES auth.users(id),
  objective_id  uuid NOT NULL REFERENCES public.learning_objectives(id),
  mastery_level numeric DEFAULT 0.0 CHECK (mastery_level >= 0 AND mastery_level <= 1),
  attempts      integer DEFAULT 0,
